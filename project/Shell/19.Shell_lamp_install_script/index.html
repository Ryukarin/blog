apr(){

	CUR_PACKAGE=`echo $APR | awk -F ".tar" '{print $1}'`
	if [ -d $SOU_PATH/$CUR_PACKAGE ];then

		./configure --prefix=$INS_PATH/apr 1>/dev/null
		if [ $? -eq 0 ];then
			make 1>/dev/null
			if [ $? -eq 0 ];then
				make install 1>/dev/null
